36      public void fieldInitializeBombs() {
37          Random random = new Random();
38          for (int x = 0; x < size; x++) {
39              for (int y = 0; y < size; y++) {
40                  fieldInitializeTiles(x, y);
41                  if (random.nextDouble() < chance) {
42                      board[x][y].setBomb();
43                      // System.out.println("pommi:" + x + y);
44                  }
45              }
46          }
47      }
48      
49      public void fieldInitializeTiles(int x, int y) {
50          board[x][y] = new Tile(x, y);
51      }
52      
53      public void forceMakeBomb(int x, int y) {
54          board[x][y].setBomb();
55      }
56  
57      public void fieldInitializeNumbers() {
58          for (int x = 0; x < size; x++) {
59              for (int y = 0; y < size; y++) {
60                  if (!board[x][y].isBomb()) {
61                      board[x][y].setValue(countNeigborBombs(board[x][y]));
62                  }
63              }
64          }
65      }
66  
67      public int countNeigborBombs(Tile tile) {
68          int value = 0;
69          for (Tile suroundingTile : getSuroundingTiles(tile)) {
70              if (suroundingTile.isBomb()) {
71                  value++;
72              }
73          }
74          return value;
75      }
76  
77      public List<Tile> getSuroundingTiles(Tile tile) {
78          ArrayList<Tile> surounders = new ArrayList<>();
79          int x = tile.getX();
80          int y = tile.getY();
81          for (int i = y - 1; i <= y + 1; i++) {
82              for (int j = x - 1; j <= x + 1; j++) {
83                  if (x == j && y == i) {
84                      continue;
85                  }
86                  if (i < 0 || i >= size) {
87                      continue;
88                  }
89                  if (j < 0 || j >= size) {
90                      continue;
91                  }
92                  surounders.add(board[j][i]);
93              }
94          }
95          return surounders;
96      }
97
